Project 1
Downloads Folder Organiser
Your Downloads folder is a mess of PDFs, images, ZIPs, and spreadsheets. Write a script that automatically sorts every file into a subdirectory by type.
os.listdiros.pathshutil.moveos.makedirs
›
What it does

Scans a folder, maps each file's extension to a category (Images, Documents, Spreadsheets, Archives, Code, Other), creates the category subfolder if it does not exist, and moves the file there. Running it twice is safe — already-sorted files are skipped.

Requirements
  1. Define a dictionary mapping categories to lists of extensions.
  2. Loop over every file in the target folder (not subdirectories).
  3. Skip files that are already inside a subdirectory.
  4. Create the destination subdirectory if it does not exist.
  5. Move the file. If a file with the same name exists at the destination, skip it.
  6. Print a summary: how many files were moved, how many skipped.
Hints
  • Use os.path.splitext() to get the extension, then .lower() it.
  • Use os.path.isfile() to skip subdirectories.
  • Use os.makedirs(dest_dir, exist_ok=True) to safely create folders.
  • Check os.path.exists(dest_path) before moving to avoid overwriting.
Real-world use: Run this on your actual Downloads folder. Or schedule it with a cron job / Task Scheduler so it runs every day automatically.
💡 Show solution 
import os
import shutil

# ── Config ──────────────────────────────────────────────
FOLDER = os.path.expanduser("~/Downloads")   # change this path

CATEGORIES = {
    "Images":       [".jpg", ".jpeg", ".png", ".gif", ".bmp", ".webp", ".svg", ".heic"],
    "Documents":    [".pdf", ".doc", ".docx", ".txt", ".md", ".ppt", ".pptx", ".odt"],
    "Spreadsheets": [".xls", ".xlsx", ".csv", ".ods"],
    "Archives":     [".zip", ".tar", ".gz", ".rar", ".7z"],
    "Code":         [".py", ".js", ".ts", ".html", ".css", ".json", ".sql", ".sh"],
    "Audio":        [".mp3", ".wav", ".flac", ".aac", ".m4a"],
    "Video":        [".mp4", ".mov", ".avi", ".mkv", ".wmv"],
}

# Build reverse map: extension → category name
EXT_MAP = {}
for category, extensions in CATEGORIES.items():
    for ext in extensions:
        EXT_MAP[ext] = category

# ── Main logic ──────────────────────────────────────────
moved = 0
skipped = 0

for name in os.listdir(FOLDER):
    src = os.path.join(FOLDER, name)

    # Skip directories (already-organised subfolders, etc.)
    if not os.path.isfile(src):
        continue

    _, ext = os.path.splitext(name)
    category = EXT_MAP.get(ext.lower(), "Other")

    dest_dir = os.path.join(FOLDER, category)
    os.makedirs(dest_dir, exist_ok=True)

    dest = os.path.join(dest_dir, name)

    if os.path.exists(dest):
        print(f"  skip (exists): {name}")
        skipped += 1
        continue

    shutil.move(src, dest)
    print(f"  moved → {category}: {name}")
    moved += 1

print(f"\nDone. {moved} moved, {skipped} skipped.")

Project 2
Multi-CSV Combiner
You have a folder (and subfolders) full of monthly sales CSVs with the same columns. Write a script that finds them all, merges them into one file, and adds a "source_file" column so you know where each row came from.
glob / rglobcsv.DictReadercsv.DictWriteros.path
›
What it does

Recursively finds all .csv files under a root folder, reads each one with csv.DictReader, adds a source_file field to every row, and writes everything into a single combined.csv. The header is written only once. Files are processed in sorted order for reproducibility.

Requirements
  1. Use glob or pathlib.rglob to find all CSVs recursively.
  2. Skip the output file itself if it already exists in the folder.
  3. Read each CSV with csv.DictReader.
  4. Add a source_file key to each row (just the filename, not the full path).
  5. Write all rows to combined.csv using csv.DictWriter.
  6. Print how many rows and files were processed.
Hints
  • Get the fieldnames from the first file's DictReader.fieldnames and append "source_file".
  • Use newline="" when opening CSV files to avoid blank rows on Windows.
  • Sort the file list: sorted(glob.glob(...)) for consistent output.
  • Use os.path.basename(filepath) for the source column value.
Real-world use: Monthly exports from accounting software, bank statements, regional sales reports — any situation where the same schema appears across many files.
💡 Show solution 
import csv
import glob
import os

# ── Config ──────────────────────────────────────────────
ROOT_FOLDER = "./sales_data"          # folder containing the CSVs
OUTPUT_FILE = os.path.join(ROOT_FOLDER, "combined.csv")

# ── Find all CSVs ────────────────────────────────────────
all_files = sorted(glob.glob(os.path.join(ROOT_FOLDER, "**", "*.csv"), recursive=True))

# Exclude the output file if it already exists there
all_files = [f for f in all_files if os.path.abspath(f) != os.path.abspath(OUTPUT_FILE)]

if not all_files:
    print("No CSV files found.")
    exit()

# ── Determine fieldnames from the first file ─────────────
with open(all_files[0], newline="", encoding="utf-8") as f:
    sample = csv.DictReader(f)
    fieldnames = sample.fieldnames + ["source_file"]

# ── Combine ──────────────────────────────────────────────
total_rows = 0

with open(OUTPUT_FILE, "w", newline="", encoding="utf-8") as out_f:
    writer = csv.DictWriter(out_f, fieldnames=fieldnames)
    writer.writeheader()

    for filepath in all_files:
        source_name = os.path.basename(filepath)
        try:
            with open(filepath, newline="", encoding="utf-8") as in_f:
                reader = csv.DictReader(in_f)
                for row in reader:
                    row["source_file"] = source_name
                    writer.writerow(row)
                    total_rows += 1
            print(f"  ✓ {source_name}")
        except Exception as e:
            print(f"  ✗ {source_name}: {e}")

print(f"\nCombined {total_rows} rows from {len(all_files)} files → {OUTPUT_FILE}")

Project 3
Config-Driven File Backup
Write a backup script that reads its settings from a JSON config file, respects a secret backup destination from an environment variable, and only copies files that have changed since the last backup.
json.loados.environshutil.copy2os.walkos.path.getmtime
›
What it does

Reads backup_config.json for the list of source folders and file extensions to include. Reads the backup destination from the BACKUP_DEST environment variable. Walks each source folder, mirrors the directory structure in the destination, and copies only files that are newer than the existing backup copy (or don't exist yet).

Requirements
  1. Load config from backup_config.json (keys: sources, include_extensions).
  2. Read BACKUP_DEST from os.environ — raise a clear error if missing.
  3. For each source folder, walk with os.walk.
  4. Skip files whose extension is not in include_extensions.
  5. Compute the destination path by mirroring the relative path under BACKUP_DEST.
  6. Copy only if the source is newer than the destination (os.path.getmtime).
  7. Print a summary: copied vs up-to-date.
Example config file
{
  "sources": [
    "/home/alice/projects",
    "/home/alice/documents"
  ],
  "include_extensions": [".py", ".csv", ".json", ".txt", ".md"]
}
Hints
  • Use os.path.relpath(src_file, source_root) to get the relative path.
  • Join it with os.path.join(BACKUP_DEST, rel_path) for the destination.
  • Use os.makedirs(os.path.dirname(dest), exist_ok=True) before copying.
  • Compare os.path.getmtime(src) > os.path.getmtime(dest) to detect changes.
Real-world use: Run this nightly via Task Scheduler (Windows) or cron (Linux/Mac) to keep an incremental backup of your work files on an external drive or network share.
💡 Show solution 
import json
import os
import shutil

# ── Load config ──────────────────────────────────────────
config_path = os.path.join(os.path.dirname(__file__), "backup_config.json")
with open(config_path, encoding="utf-8") as f:
    config = json.load(f)

sources    = config["sources"]
include_ext = set(ext.lower() for ext in config["include_extensions"])

# ── Read destination from environment ───────────────────
backup_dest = os.environ.get("BACKUP_DEST")
if not backup_dest:
    raise EnvironmentError(
        "BACKUP_DEST environment variable is not set. "
        "Set it to the backup root directory before running."
    )

# ── Backup loop ──────────────────────────────────────────
copied = 0
up_to_date = 0

for source_root in sources:
    if not os.path.isdir(source_root):
        print(f"WARNING: source not found, skipping: {source_root}")
        continue

    for dirpath, dirnames, filenames in os.walk(source_root):
        for filename in filenames:
            _, ext = os.path.splitext(filename)
            if ext.lower() not in include_ext:
                continue

            src_file = os.path.join(dirpath, filename)
            rel_path = os.path.relpath(src_file, os.path.dirname(source_root))
            dest_file = os.path.join(backup_dest, rel_path)

            dest_dir = os.path.dirname(dest_file)
            os.makedirs(dest_dir, exist_ok=True)

            # Copy only if source is newer or destination doesn't exist
            if (not os.path.exists(dest_file) or
                    os.path.getmtime(src_file) > os.path.getmtime(dest_file)):
                shutil.copy2(src_file, dest_file)
                print(f"  copied: {rel_path}")
                copied += 1
            else:
                up_to_date += 1

print(f"\nBackup complete. {copied} copied, {up_to_date} already up-to-date.")

Project Scaffolder

pathlibos.makedirsjson.dumpos.environsys.argv

What it does

Takes a project name as a command-line argument (python scaffold.py my_project). Creates a directory tree: data/raw, data/processed, notebooks, src, output. Writes a config.json template, a .gitignore that excludes .env and data folders, a README.md, and a skeleton src/pipeline.py. Reads the author name from the GIT_AUTHOR_NAME env variable if available.

Requirements
  1. Read the project name from sys.argv[1]. Print usage and exit if missing.
  2. Refuse to create if a folder with that name already exists.
  3. Use pathlib.Path and the / operator throughout.
  4. Create all subdirectories with mkdir(parents=True).
  5. Write config.json with json.dump and indent=4.
  6. Write .gitignore, README.md, and src/pipeline.py as text files.
  7. Print the created tree at the end.
Hints
  • Path(name).exists() to check if the folder already exists.
  • Path.write_text(content, encoding="utf-8") for writing template files.
  • os.environ.get("GIT_AUTHOR_NAME", "Your Name") for the author fallback.
Real-world use: Instantly get a clean, consistent starting structure for every new analysis project. Add it to your PATH and call it from anywhere.

import json
import os
import sys
from pathlib import Path

# ── Args ─────────────────────────────────────────────────
if len(sys.argv) < 2:
    print("Usage: python scaffold.py ")
    sys.exit(1)

name = sys.argv[1]
root = Path(name)
author = os.environ.get("GIT_AUTHOR_NAME", "Your Name")

if root.exists():
    print(f"Error: '{name}' already exists.")
    sys.exit(1)

# ── Directory structure ───────────────────────────────────
dirs = [
    root / "data" / "raw",
    root / "data" / "processed",
    root / "notebooks",
    root / "src",
    root / "output",
]
for d in dirs:
    d.mkdir(parents=True)

# ── config.json ───────────────────────────────────────────
config = {
    "project": name,
    "author": author,
    "input_dir": "data/raw",
    "output_dir": "data/processed",
    "log_level": "INFO"
}
(root / "config.json").write_text(
    json.dumps(config, indent=4), encoding="utf-8"
)

# ── .gitignore ────────────────────────────────────────────
(root / ".gitignore").write_text(
    ".env\n"
    "data/raw/\n"
    "data/processed/\n"
    "output/\n"
    "__pycache__/\n"
    "*.pyc\n"
    ".DS_Store\n",
    encoding="utf-8"
)

# ── README.md ─────────────────────────────────────────────
(root / "README.md").write_text(
    f"# {name}\n\n"
    f"**Author:** {author}\n\n"
    "## Structure\n\n"
    "```\n"
    f"{name}/\n"
    "├── data/\n"
    "│   ├── raw/          # original input files (not in git)\n"
    "│   └── processed/    # cleaned output files\n"
    "├── notebooks/        # exploratory analysis\n"
    "├── src/\n"
    "│   └── pipeline.py   # main processing script\n"
    "├── output/           # final deliverables\n"
    "├── config.json       # runtime settings\n"
    "└── .gitignore\n"
    "```\n\n"
    "## Usage\n\n"
    "```bash\n"
    "python src/pipeline.py\n"
    "```\n",
    encoding="utf-8"
)

# ── src/pipeline.py ───────────────────────────────────────
(root / "src" / "pipeline.py").write_text(
    '"""Main pipeline for ' + name + '."""\n'
    "import json\n"
    "import os\n"
    "from pathlib import Path\n\n"
    "ROOT = Path(__file__).parent.parent\n\n"
    'with open(ROOT / "config.json", encoding="utf-8") as f:\n'
    "    config = json.load(f)\n\n"
    "input_dir  = ROOT / config[\"input_dir\"]\n"
    "output_dir = ROOT / config[\"output_dir\"]\n"
    "output_dir.mkdir(parents=True, exist_ok=True)\n\n"
    "def main():\n"
    "    files = sorted(input_dir.glob(\"*.csv\"))\n"
    "    print(f\"Found {len(files)} input file(s).\")\n"
    "    for f in files:\n"
    "        print(f\"  processing {f.name}...\")\n"
    "        # TODO: add your processing logic here\n\n"
    'if __name__ == "__main__":\n'
    "    main()\n",
    encoding="utf-8"
)

# ── Summary ───────────────────────────────────────────────
print(f"\n✓ Project '{name}' created:\n")
for path in sorted(root.rglob("*")):
    indent = "  " * (len(path.relative_to(root).parts) - 1)
    icon = "📁" if path.is_dir() else "📄"
    print(f"  {indent}{icon} {path.name}")

Project 5
Log File Analyser
You have a folder of application log files (.log). Write a script that scans all logs, counts ERROR and WARNING lines, extracts unique error messages, and writes a clean summary report to a text file.
globtext file readingos.pathfile writingpathlib
›
What it does

Finds all .log files in a target folder. For each file, reads line-by-line (memory-efficient), counts ERROR and WARNING occurrences, and collects unique error messages. Writes a summary_report.txt with per-file stats and the top 10 most frequent error messages overall.

Requirements
  1. Use glob or pathlib.rglob to find all .log files.
  2. Read each file line-by-line (do not load the whole file at once).
  3. Count lines containing "ERROR" and "WARNING" (case-sensitive or insensitive — your choice).
  4. Collect all unique error messages in a Counter for frequency analysis.
  5. Write summary_report.txt with: per-file counts, grand total, and top 10 errors.
  6. Also print the summary to the terminal.
Hints
  • Use from collections import Counter for frequency counting.
  • Strip error lines: line.strip() before adding to the counter.
  • Use counter.most_common(10) to get the top 10 entries.
  • Open the output with mode "w" and write lines manually with f.write().
Real-world use: After deploying an application, run this on your server's log directory to instantly see what errors are happening and how often — without opening each file manually.
💡 Show solution 
import glob
import os
from collections import Counter
from pathlib import Path

# ── Config ──────────────────────────────────────────────
LOGS_FOLDER = "./logs"
REPORT_FILE = os.path.join(LOGS_FOLDER, "summary_report.txt")

# ── Find log files ───────────────────────────────────────
log_files = sorted(glob.glob(os.path.join(LOGS_FOLDER, "**", "*.log"), recursive=True))

if not log_files:
    print("No .log files found.")
    exit()

# ── Analyse ───────────────────────────────────────────────
error_counter = Counter()
file_stats = []   # list of (filename, error_count, warning_count)

grand_errors   = 0
grand_warnings = 0

for filepath in log_files:
    errors   = 0
    warnings = 0

    try:
        with open(filepath, encoding="utf-8", errors="replace") as f:
            for line in f:
                if "ERROR" in line:
                    errors += 1
                    error_counter[line.strip()] += 1
                elif "WARNING" in line:
                    warnings += 1
    except Exception as e:
        print(f"Could not read {filepath}: {e}")
        continue

    filename = os.path.basename(filepath)
    file_stats.append((filename, errors, warnings))
    grand_errors   += errors
    grand_warnings += warnings

# ── Build report ─────────────────────────────────────────
lines = []
lines.append("=" * 60)
lines.append("LOG ANALYSIS REPORT")
lines.append("=" * 60)
lines.append(f"Files scanned : {len(file_stats)}")
lines.append(f"Total ERRORs  : {grand_errors}")
lines.append(f"Total WARNINGs: {grand_warnings}")
lines.append("")
lines.append("Per-file breakdown:")
lines.append("-" * 40)

for fname, errs, warns in file_stats:
    lines.append(f"  {fname:<30} ERR:{errs:>4}  WARN:{warns:>4}")

lines.append("")
lines.append("Top 10 most frequent ERROR lines:")
lines.append("-" * 40)

for i, (msg, count) in enumerate(error_counter.most_common(10), 1):
    # Truncate long lines for readability
    short = msg[:80] + "..." if len(msg) > 80 else msg
    lines.append(f"  {i:>2}. [{count:>4}x]  {short}")

lines.append("=" * 60)

report_text = "\n".join(lines)

# ── Write to file and print ──────────────────────────────
Path(LOGS_FOLDER).mkdir(parents=True, exist_ok=True)

with open(REPORT_FILE, "w", encoding="utf-8") as f:
    f.write(report_text + "\n")

print(report_text)
print(f"\nReport saved → {REPORT_FILE}")
