Chapter 1
Reading Excel Files

For reading existing Excel files, pandas with openpyxl as the engine is the standard approach. For creating or editing formatted Excel files, use openpyxl directly.

pip install openpyxl xlsxwriter

Pandas read_excel

import pandas as pd

# Simple read
df = pd.read_excel('sales_report.xlsx')

# Common situations
df = pd.read_excel(
    'sales_report.xlsx',
    sheet_name='Q1 Data',  # by name or 0-based index
    header=2,              # header is in row 3 (0-indexed)
    usecols='B:G',         # only columns B through G
    skiprows=[0, 1],       # skip logo/title rows
    na_values=['N/A', '-']
)

# All sheets at once
all_sheets = pd.read_excel('report.xlsx', sheet_name=None)  # dict
for name, df in all_sheets.items():
    print(f"Sheet '{name}': {df.shape}")

Reading with openpyxl directly

Use openpyxl when you need cell-level access (colours, formulas, merged cells, etc.).

from openpyxl import load_workbook

wb = load_workbook('report.xlsx', data_only=True)  # data_only=True reads cell values, not formulas
ws = wb.active           # first sheet
# or
ws = wb['Sheet1']        # by name
ws = wb.worksheets[1]    # by index

# Read a single cell
val = ws['B3'].value
val = ws.cell(row=3, column=2).value

# Iterate rows
for row in ws.iter_rows(min_row=2, values_only=True):
    print(row)  # tuple of cell values

# All data into a list of dicts
headers = [ws.cell(1, c).value for c in range(1, ws.max_column+1)]
data = []
for row in ws.iter_rows(min_row=2, values_only=True):
    data.append(dict(zip(headers, row)))

df = pd.DataFrame(data)

Writing with openpyxl
from openpyxl import Workbook

wb = Workbook()
ws = wb.active
ws.title = 'Sales Data'

# Write headers
headers = ['Rep', 'Region', 'Sales', 'Month']
for col, h in enumerate(headers, start=1):
    ws.cell(row=1, column=col, value=h)

# Write rows from a DataFrame
import pandas as pd
df = pd.read_csv('sales.csv')

for r_idx, row in enumerate(df.itertuples(index=False), start=2):
    for c_idx, val in enumerate(row, start=1):
        ws.cell(row=r_idx, column=c_idx, value=val)

wb.save('output.xlsx')

The faster pandas route

When you don't need custom formatting, use pandas to write the data, then reopen with openpyxl for formatting:

with pd.ExcelWriter('output.xlsx', engine='openpyxl') as writer:

    df_summary.to_excel(writer, sheet_name='Summary', index=False)

Chapter 3
Cell Formatting

openpyxl lets you set fonts, fills, borders, alignment, and number formats on any cell.

ABCD
1RepRegionSalesMonth
2AliceNorth12,400Jan
3BobSouth15,800Jan
4CarolNorth9,700Feb
5DaveEast18,300Feb
6TOTAL56,200
from openpyxl.styles import (
    Font, PatternFill, Alignment, Border, Side, numbers
)
from openpyxl.utils import get_column_letter

# Font
ws['A1'].font = Font(bold=True, size=12, color='FFFFFF', name='Calibri')

# Fill (background colour)
from openpyxl.styles import PatternFill
green_fill  = PatternFill(fill_type='solid', fgColor='1F7244')
alt_fill    = PatternFill(fill_type='solid', fgColor='F2F9F2')
yellow_fill = PatternFill(fill_type='solid', fgColor='FFF3CD')

ws['A1'].fill = green_fill

# Apply to header row
for cell in ws[1]:
    cell.fill   = green_fill
    cell.font   = Font(bold=True, color='FFFFFF')
    cell.alignment = Alignment(horizontal='center', vertical='center')

# Alternating row fill
for row_idx in range(2, ws.max_row + 1):
    fill = alt_fill if row_idx % 2 == 0 else PatternFill()
    for cell in ws[row_idx]:
        cell.fill = fill

# Number format
for row in ws.iter_rows(min_row=2, min_col=3, max_col=3):
    for cell in row:
        cell.number_format = '#,##0'        # 12,400
        # or: '#,##0.00'  / '$#,##0.00' / '0%'

# Borders
thin = Side(style='thin', color='D6D6D6')
border = Border(left=thin, right=thin, top=thin, bottom=thin)
for row in ws.iter_rows():
    for cell in row:
        cell.border = border

# Row height
ws.row_dimensions[1].height = 22

wb.save('formatted.xlsx')
Hex colours in openpyxl openpyxl uses hex without the leading #. So CSS #1F7244 becomes 1F7244.
Chapter 4
Tables & Conditional Formatting

Excel Tables (ListObject)

Converting a range to an Excel Table enables sorting, filtering, and auto-extending formulas in Excel itself.

from openpyxl.worksheet.table import Table, TableStyleInfo

table = Table(
    displayName='SalesTable',
    ref=f'A1:D{ws.max_row}'   # full data range
)
style = TableStyleInfo(
    name='TableStyleMedium9',   # Excel's built-in style names
    showFirstColumn=False,
    showLastColumn=False,
    showRowStripes=True,
    showColumnStripes=False
)
table.tableStyleInfo = style
ws.add_table(table)
wb.save('with_table.xlsx')

Conditional Formatting

from openpyxl.formatting.rule import (
    ColorScaleRule, DataBarRule, CellIsRule, FormulaRule
)
from openpyxl.styles import PatternFill, Font

# Colour scale: low=red, mid=yellow, high=green
color_scale = ColorScaleRule(
    start_type='min',  start_color='FF4444',
    mid_type='percentile', mid_value=50, mid_color='FFFF44',
    end_type='max',    end_color='44BB44'
)
ws.conditional_formatting.add('C2:C100', color_scale)

# Data bar
data_bar = DataBarRule(start_type='min', start_value=0,
                       end_type='max', end_value=None, color='638EC6')
ws.conditional_formatting.add('C2:C100', data_bar)

# Highlight cells above threshold
red_fill = PatternFill(fill_type='solid', fgColor='FFE0E0')
above_rule = CellIsRule(operator='greaterThan', formula=['15000'], fill=red_fill)
ws.conditional_formatting.add('C2:C100', above_rule)

wb.save('conditional.xlsx')
Chapter 5
Embedded Charts

openpyxl can insert native Excel charts that the user can interact with after opening the file.

from openpyxl.chart import BarChart, LineChart, Reference

# --- Bar chart ---
bar = BarChart()
bar.type   = 'col'       # 'col' = vertical bars, 'bar' = horizontal
bar.style  = 10
bar.title  = 'Sales by Region'
bar.y_axis.title = 'Sales ($)'
bar.x_axis.title = 'Region'

# Data reference: rows 1-5, column C (sales values)
data = Reference(ws, min_col=3, min_row=1, max_row=5)
bar.add_data(data, titles_from_data=True)

# Category labels: column A (region names)
cats = Reference(ws, min_col=1, min_row=2, max_row=5)
bar.set_categories(cats)

bar.shape = 4              # rounded corners
bar.width  = 15            # chart width in cm
bar.height = 10

ws.add_chart(bar, 'F2')    # top-left corner of chart placement
wb.save('with_chart.xlsx')
# --- Line chart ---
line = LineChart()
line.title  = 'Monthly Sales Trend'
line.y_axis.title = 'Sales'
line.x_axis.title = 'Month'
line.style  = 10
line.width  = 20
line.height = 12

# Multiple series (one per region)
for col_idx in range(2, 5):   # columns B, C, D = regions
    data = Reference(ws, min_col=col_idx, min_row=1, max_row=13)
    line.add_data(data, titles_from_data=True)

months_ref = Reference(ws, min_col=1, min_row=2, max_row=13)
line.set_categories(months_ref)
ws.add_chart(line, 'A16')
wb.save('line_chart.xlsx')
Available chart types in openpyxl BarChart, LineChart, PieChart, DoughnutChart, AreaChart, ScatterChart, BubbleChart, RadarChart, StockChart. All live in openpyxl.chart.

Chapter 7
Full Report Automation

This is the end goal — a Python script that runs on new data and produces a complete, formatted Excel report with summary stats, a data table, conditional formatting, and an embedded chart.

"""
generate_sales_report.py
Produces a formatted Excel report from a CSV file.
Usage:  python generate_sales_report.py sales_data.csv
"""
import sys
from pathlib import Path
from datetime import date

import pandas as pd
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side, numbers as xl_num
from openpyxl.utils import get_column_letter
from openpyxl.chart import BarChart, Reference
from openpyxl.worksheet.table import Table, TableStyleInfo
from openpyxl.formatting.rule import ColorScaleRule

# ── 1. Load & aggregate ────────────────────────────────────────────────────
csv_path = Path(sys.argv[1]) if len(sys.argv) > 1 else Path('sales.csv')
df = pd.read_csv(csv_path, parse_dates=['date'])
df.columns = df.columns.str.strip().str.lower()

summary = (
    df.groupby('region')
      .agg(total_sales=('sales','sum'), deals=('sales','count'), avg_deal=('sales','mean'))
      .round(2)
      .reset_index()
)

# ── 2. Create workbook ─────────────────────────────────────────────────────
wb = Workbook()

# ── Sheet 1: Raw Data ──────────────────────────────────────────────────────
ws_raw = wb.active
ws_raw.title = 'Raw Data'
ws_raw.sheet_properties.tabColor = '7F8C8D'
ws_raw.freeze_panes = 'A2'

raw_headers = list(df.columns)
for c, h in enumerate(raw_headers, 1):
    cell = ws_raw.cell(1, c, h.replace('_', ' ').title())
    cell.font  = Font(bold=True, color='FFFFFF', size=10)
    cell.fill  = PatternFill('solid', fgColor='2C3E50')
    cell.alignment = Alignment(horizontal='center')

for r, row in enumerate(df.itertuples(index=False), 2):
    fill = PatternFill('solid', fgColor='F8F9FA') if r % 2 == 0 else PatternFill()
    for c, val in enumerate(row, 1):
        cell = ws_raw.cell(r, c, val)
        cell.fill = fill
        if isinstance(val, float):
            cell.number_format = '#,##0.00'

# Auto column width
for col in ws_raw.columns:
    w = max(len(str(cell.value or '')) for cell in col)
    ws_raw.column_dimensions[get_column_letter(col[0].column)].width = min(w + 4, 35)

# Table
tbl = Table(displayName='RawSales', ref=f'A1:{get_column_letter(len(raw_headers))}{len(df)+1}')
tbl.tableStyleInfo = TableStyleInfo(name='TableStyleMedium2', showRowStripes=True)
ws_raw.add_table(tbl)

# ── Sheet 2: Summary ───────────────────────────────────────────────────────
ws_sum = wb.create_sheet('Summary')
ws_sum.sheet_properties.tabColor = '1F7244'
ws_sum.freeze_panes = 'A2'

# Title
ws_sum['A1'] = f'Sales Summary Report — {date.today().strftime("%d %b %Y")}'
ws_sum['A1'].font = Font(bold=True, size=14, color='1A252F')
ws_sum.merge_cells('A1:D1')
ws_sum.row_dimensions[1].height = 26
ws_sum['A1'].alignment = Alignment(horizontal='left', vertical='center')

sum_headers = ['Region', 'Total Sales', 'Deals', 'Avg Deal']
for c, h in enumerate(sum_headers, 1):
    cell = ws_sum.cell(2, c, h)
    cell.font  = Font(bold=True, color='FFFFFF', size=10)
    cell.fill  = PatternFill('solid', fgColor='1F7244')
    cell.alignment = Alignment(horizontal='center')

thin = Side(style='thin', color='D0D0D0')
border = Border(left=thin, right=thin, top=thin, bottom=thin)

for r, row in enumerate(summary.itertuples(index=False), 3):
    bg = 'F2F9F2' if r % 2 == 0 else 'FFFFFF'
    ws_sum.cell(r, 1, row.region).fill = PatternFill('solid', fgColor=bg)
    c2 = ws_sum.cell(r, 2, row.total_sales)
    c2.number_format = '$#,##0.00'
    c2.fill = PatternFill('solid', fgColor=bg)
    ws_sum.cell(r, 3, row.deals).fill = PatternFill('solid', fgColor=bg)
    c4 = ws_sum.cell(r, 4, row.avg_deal)
    c4.number_format = '$#,##0.00'
    c4.fill = PatternFill('solid', fgColor=bg)
    for c in ws_sum[r]:
        c.border = border

# Totals row
tr = len(summary) + 3
ws_sum.cell(tr, 1, 'TOTAL').font = Font(bold=True)
total_cell = ws_sum.cell(tr, 2, summary['total_sales'].sum())
total_cell.number_format = '$#,##0.00'
total_cell.font = Font(bold=True)
total_cell.fill = PatternFill('solid', fgColor='FFF3CD')
for c in ws_sum[tr]:
    c.fill = PatternFill('solid', fgColor='FFF3CD')
    c.font = Font(bold=True)

# Color scale on Total Sales column
ws_sum.conditional_formatting.add(
    f'B3:B{tr-1}',
    ColorScaleRule(start_type='min', start_color='FF9999',
                   end_type='max',   end_color='44BB44')
)

# Auto widths
for col in ws_sum.columns:
    w = max(len(str(cell.value or '')) for cell in col)
    ws_sum.column_dimensions[get_column_letter(col[0].column)].width = min(w + 4, 30)

# ── Chart on Summary sheet ─────────────────────────────────────────────────
bar = BarChart()
bar.type = 'col'; bar.style = 10
bar.title = 'Total Sales by Region'
bar.y_axis.title = 'Sales ($)'
bar.x_axis.title = 'Region'
bar.width = 16; bar.height = 10

data_ref = Reference(ws_sum, min_col=2, min_row=2, max_row=len(summary)+2)
bar.add_data(data_ref, titles_from_data=True)
cats_ref = Reference(ws_sum, min_col=1, min_row=3, max_row=len(summary)+2)
bar.set_categories(cats_ref)

ws_sum.add_chart(bar, f'F2')

# ── Save ───────────────────────────────────────────────────────────────────
out_path = csv_path.with_name(csv_path.stem + '_report.xlsx')
wb.save(out_path)
print(f"Report saved: {out_path}")
xlsxwriter vs openpyxl — quick comparison
  • openpyxl — read AND write; modify existing files; best for most tasks
  • xlsxwriter — write-only but faster for very large files; better sparklines support; use with pd.ExcelWriter(engine='xlsxwriter')
